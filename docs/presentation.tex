\begin{frame}{Project Idea}
  \textbf{Goal:} price American-style financial options with a binomial tree.

  \vspace{0.3cm}

  \begin{itemize}
    \item Input: option parameters from \texttt{data/options.csv}
    \item Optional API update: latest AAPL spot price from Alpha Vantage
    \item API key can be stored locally in \texttt{.alphavantage\_key}
    \item If the API fails, the program falls back to CSV values
    \item Output: American price, European price, early exercise premium
    \item Extra: volatility analysis and a small SVG tree visualization
  \end{itemize}
\end{frame}

\begin{frame}{Model And Formula}
  Cox-Ross-Rubinstein binomial model:

  \[
    u = e^{\sigma\sqrt{\Delta t}},
    \quad
    d = \frac{1}{u},
    \quad
    p = \frac{e^{r\Delta t} - d}{u-d}
  \]

  Continuation value:

  \[
    V_{\text{cont}} = e^{-r\Delta t}
    \left(pV_{\text{up}} + (1-p)V_{\text{down}}\right)
  \]

  American exercise decision:

  \[
    V = \max(V_{\text{cont}}, V_{\text{exercise}})
  \]
\end{frame}

\begin{frame}{C++ Features Used}
  \begin{itemize}
    \item \texttt{struct Option}: stores spot, strike, rate, volatility, maturity
    \item \texttt{enum}: call/put and European/American
    \item \texttt{std::vector<Option>}: stores all options loaded from CSV
    \item \texttt{std::vector<double>}: stores one binomial tree level
    \item \texttt{std::ifstream}, \texttt{std::ofstream}: CSV input and SVG output
    \item \texttt{std::stringstream}: simple CSV parsing
    \item \texttt{std::max}: payoff and early exercise decision
    \item Not used: threads, \texttt{optional}, \texttt{variant}; not needed for this small deterministic model
  \end{itemize}
\end{frame}

\begin{frame}{Output And Pipeline}
  \textbf{Pipeline:}

  \begin{enumerate}
    \item CSV fallback or AAPL API data
    \item Store option parameters in \texttt{std::vector<Option>}
    \item Run binomial pricing with backward induction
    \item Print tables and generate \texttt{binomial\_tree.svg}
  \end{enumerate}

  \vspace{0.3cm}

  The program prints:
  \begin{itemize}
    \item American vs European option prices
    \item Early exercise premium
    \item Price stability for different step counts
    \item Volatility analysis
    \item A 3-step tree preview and \texttt{binomial\_tree.svg}
  \end{itemize}
\end{frame}
